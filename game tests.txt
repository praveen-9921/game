    <script>
        let balance = 1000;
        let countdown = 120;
        let userBets = [];

        function updateCountdown() {
            let minutes = Math.floor(countdown / 60);
            let seconds = countdown % 60;
            document.getElementById("countdown").textContent = `${minutes}:${seconds < 10 ? '0' : ''}${seconds}`;
            if (countdown > 0) {
                countdown--;
                setTimeout(updateCountdown, 1000);
            } else {
                showResults();
            }
        }
        updateCountdown();

        function placeBet(choice) {
            let betAmount = parseInt(prompt("Enter your bet amount:"));
            if (isNaN(betAmount) || betAmount <= 0 || betAmount > balance) {
                alert("Invalid bet amount!");
                return;
            }
            userBets.push({ choice, betAmount });
            balance -= betAmount;
            document.getElementById("balance").textContent = balance;
            alert(`Bet placed on ${choice} with ₹${betAmount}`);
        }

        function showResults() {
            const colors = ['red', 'green', 'violet'];
            const randomColor = colors[Math.floor(Math.random() * colors.length)];
            const randomNumber = Math.floor(Math.random() * 10);
            let totalWinnings = 0;
            
            userBets.forEach(bet => {
                if (bet.choice === randomColor || bet.choice === randomNumber) {
                    totalWinnings += bet.betAmount * 2;
                }
            });
            
            balance += totalWinnings;
            document.getElementById("balance").textContent = balance;
            document.getElementById("result").textContent = `Winning Color: ${randomColor}, Winning Number: ${randomNumber}`;
            userBets = [];
            countdown = 120;
            updateCountdown();
        }

        function recharge() {
            balance += 500;
            document.getElementById("balance").textContent = balance;
            alert("Balance Recharged!");

            alert("Pick a color or number and place a bet. When the timer ends, the winning color and number will be revealed. If your choice matches, you win double your bet!");
        }
